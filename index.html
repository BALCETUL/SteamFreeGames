  <div id="steamfree-widget">Загрузка...</div>
  <p style="margin-top:1em;font-size:12px;color:#888;text-align:right;">
    <a href="https://github.com/BALCETUL/SteamFreeGames" target="_blank" style="color:#aaa;text-decoration:none;">Источник: BALCETUL</a>
  </p>
</div>

<script src="https://raw.githubusercontent.com/BALCETUL/SteamFreeGames/main/embed.js"></script>
<script>
  SteamFreeEmbed.init({
    selector: '#steamfree-widget',
    maxItems: 5,
    showImages: true,
    width: '100%'
  });
</script>
